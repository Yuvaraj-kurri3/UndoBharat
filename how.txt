    <script>
        // Dark Mode Toggle
        const darkModeToggle = document.getElementById('darkModeToggle');
        const mobileDarkModeToggle = document.getElementById('mobileDarkModeToggle');
        const html = document.documentElement;

        // Check for saved user preference or use system preference
        if (localStorage.getItem('darkMode') === 'true' || 
            (!localStorage.getItem('darkMode') && window.matchMedia('(prefers-color-scheme: dark)').matches)) {
            html.classList.add('dark');
            darkModeToggle.checked = true;
            mobileDarkModeToggle.checked = true;
        }

        // Toggle dark mode
        function toggleDarkMode() {
            html.classList.toggle('dark');
            localStorage.setItem('darkMode', html.classList.contains('dark'));
        }

        darkModeToggle.addEventListener('change', toggleDarkMode);
        mobileDarkModeToggle.addEventListener('change', toggleDarkMode);

        // Mobile Menu Toggle
        const hamburger = document.getElementById('hamburger');
        const mobileMenu = document.getElementById('mobileMenu');

        hamburger.addEventListener('click', () => {
            hamburger.classList.toggle('hamburger-active');
            mobileMenu.classList.toggle('hidden');
        });

        // User/Admin Flow Toggle
        const userBtn = document.getElementById('userBtn');
        const adminBtn = document.getElementById('adminBtn');
        const userFlow = document.getElementById('userFlow');
        const adminFlow = document.getElementById('adminFlow');

        function showFlow(flowToShow, flowToHide, activeBtn, inactiveBtn) {
            // Hide all flows first
            document.querySelectorAll('[id$="Flow"]').forEach(flow => {
                flow.classList.add('hidden');
            });
            
            // Show the selected flow
            flowToShow.classList.remove('hidden');
            
            // Update button styles
            activeBtn.classList.remove('bg-gray-500');
            activeBtn.classList.add('bg-blue-500');
            inactiveBtn.classList.remove('bg-blue-500');
            inactiveBtn.classList.add('bg-gray-500');
            
            // Animate the cards
            const cards = flowToShow.querySelectorAll('.flow-card');
            cards.forEach(card => {
                card.classList.remove('show');
                setTimeout(() => card.classList.add('show'), 50);
            });
        }

        userBtn.addEventListener('click', () => {
            showFlow(userFlow, adminFlow, userBtn, adminBtn);
        });

        adminBtn.addEventListener('click', () => {
            showFlow(adminFlow, userFlow, adminBtn, userBtn);
        });

        // Initialize animation for user flow cards
        document.addEventListener('DOMContentLoaded', () => {
            const cards = userFlow.querySelectorAll('.flow-card');
            cards.forEach(card => {
                setTimeout(() => card.classList.add('show'), 100);
            });
        });
    </script>
